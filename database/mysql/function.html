    # 函数

    1. CONCAT 用来拼接字符串

      ### 拼接字符串
select concat('my','sql')
### 结果
mysql
      1
      2
      3
      4

    2. LEFT 从左边截取字符

      ### 从左往右开始截取字符串长度为5
select left('woshinibaba',5)
### 结果
woshi
      1
      2
      3
      4

    3. RIGHT 从右边截取字符

      ### 从右往左截取字符串长度为5
select right('woshinibaba',5)
### 结果
ibaba
      1
      2
      3
      4

    4. UPPER 变为大写

      ### 将所有字符串变为大写
select upper('woshinibaba')
### 结果
WOSHINIBABA
      1
      2
      3
      4

    5. LOWER 变为小写

      ### 将字符串变成小写
select lower('WOSHINIBABA')
### 结果
woshinibaba
      1
      2
      3
      4

    6. SUBSTRING 截取字符串

      ### 截取字符串
select substring('woshinibaba',5)
### 结果
nibaba
      1
      2
      3
